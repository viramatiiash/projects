// 1st variant
function isSubsetOf(s1, s2) {
  const hasAllValues = [...s1].every((value) => s2.has(value));
  if (hasAllValues && s1.size <= s2.size) {
    return true;
  } else if (s2.size === 0) {
    return false;
  }
  return false;
}
